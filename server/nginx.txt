server {
    if ($host = www.ojest.pl) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = ojest.pl) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    listen [::]:80;
    server_name ojest.pl www.ojest.pl;

    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    location / {
        return 301 https://$host$request_uri;
    }




}


server {
    listen 443 ssl;
    server_name ojest.pl www.ojest.pl;
    ssl_certificate /etc/letsencrypt/live/ojest.pl/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/ojest.pl/privkey.pem; # managed by Certbot


    # WebSocket (Socket.io)
    location /socket.io/ {
        proxy_pass http://localhost:5000/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }


    # Backend
    location /api/ {
        proxy_pass http://localhost:5000/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # Frontend
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }




}
